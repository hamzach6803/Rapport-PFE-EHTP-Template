}{}

    BoldItalicFont=*-BoldItalic

      \newfontfamily\arabicfont[Script=Arabic,Scale=1.1]{Arial}
    }{%
      \newfontfamily\arabicfont[Script=Arabic,Scale=1.1]{Latin Modern Roman}
    }%
  }%
